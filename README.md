```bash
npm install --save taoladat
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'taoladat'
import 'taoladat/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
